import { Meta, ColorPalette, ColorItem, Title } from "@storybook/addon-docs/blocks";
import { colors } from "./colors";


<Meta title="Colors" />

<ColorPalette>
{Object.entries({...colors.palette, ...colors.backgroundColor, ...colors.textColor}).map(([key, value], index) => (
  <>
    {colors.palette[key] && <h2>Palette Colors</h2>}
    {colors.backgroundColor[key] && <h2>Background Colors</h2>}
    {colors.textColor[key] && <h2>Text Colors</h2>}
    
    <ColorItem 
      title={
        colors.palette[key] ? `Palette-${key}` :
        colors.backgroundColor[key] ? `Background-${key}` :
        colors.textColor[key] ? `Text-${key}` :
        key
      } 
      subtitle={`${colors.textColor[key] ? "text-" : "" }${key}-[value]`} 
      colors={value} 
    />
  </>
))}
</ColorPalette>
